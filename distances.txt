path(X,X,Distance,H) :- 
    H >= 0, Distance is 0.

path(X,Y,Distance,H) :- 
    H > 1, route(X,Z,Distance1),
    H1 is H-1, 
    path(Z,Y,Distance2,H1), 
    Distance is Distance1+Distance2. %Finds any path

findPath(X,Y,Distance,H) :- %Base case if theres a single path between StationX and StationY
    path(X,Y,Distance,H).

findPath(X,Y,Distance,H) :- 
    H1 is H+1,
    findPath(X,Y,Distance,H1).

pathBound(X,Y,Distance,L) :-
    route(X,Y,D),
    D < L,
    Distance is D.

pathBound(X,Y,Distance,L) :- 
    L >=0,
    route(X,Z,Distance1),
    L1 is L - Distance1,
    pathBound(Z,Y,Distance2,L1),
    Distance is Distance1 + Distance2.

distance(X,Y,Distance) :-
    findPath(X,Y,Distance,0),
    \+ pathBound(X,Y,_,Distance).
